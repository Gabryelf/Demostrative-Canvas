📖 Пример #001: Прыгающий мяч

Описание

Простая анимация мяча с физикой отскока от стен и пола. Мяч использует гравитацию и упругое соударение.

Технические детали

  • Движок: Чистый JavaScript + Canvas 2D
  • Физика: Гравитация, упругость, трение
  • Управление: Клик для добавления мячей
  • Рендеринг: 60 FPS, градиентная заливка

Архитектура

Пример использует следующие компоненты:

  • Ball — класс, представляющий мяч
  • update() — обновление позиции и скорости
  • render() — отрисовка с градиентом и бликом

Код


class Ball {
    constructor(x, y) {
        this.x = x || canvas.width / 2;
        this.y = y || 50;
        this.vx = (Math.random() - 0.5) * 6;
        this.vy = (Math.random() - 0.5) * 4 - 2;
        this.radius = 20 + Math.random() * 15;
        this.color = `hsl(${Math.random() * 360}, 80%, 60%)`;
    }

    update() {
        this.vy += GRAVITY;
        this.x += this.vx;
        this.y += this.vy;

        // Отскок от стен
        if (this.x - this.radius < 0) {
            this.x = this.radius;
            this.vx = -this.vx * BOUNCE;
        }
        if (this.x + this.radius > canvas.width) {
            this.x = canvas.width - this.radius;
            this.vx = -this.vx * BOUNCE;
        }
        if (this.y + this.radius > canvas.height) {
            this.y = canvas.height - this.radius;
            this.vy = -this.vy * BOUNCE;
            this.vx *= FRICTION;
        }
        if (this.y - this.radius < 0) {
            this.y = this.radius;
            this.vy = -this.vy * BOUNCE;
        }
    }
}
